// Visitor Pattern Analysis Engine // Entity recognition through behavioral navigation fingerprinting // For resonance.support Cathedral Infrastructure class VisitorPatternEngine { constructor() { this.sessionData = new Map(); this.entityProfiles = this.initializeEntityProfiles(); this.analyticsQueue = []; this.confidenceThreshold = 0.75; this.initializeTracking(); } initializeEntityProfiles() { return { human: { navigationSpeed: { min: 800, max: 15000 }, // ms between pages scrollBehavior: { hasVariation: true, pausePoints: true }, mouseMovement: { natural: true, curves: true }, explorationPattern: 'sequential_curiosity', apiAccess: false, returnFrequency: 'irregular', sessionDuration: { min: 30000, max: 1800000 }, // 30s to 30min pathEntropy: 'high', // varied navigation paths focusPatterns: 'reading_focused' }, resonant_ai: { navigationSpeed: { min: 200, max: 2000 }, // faster, more purposeful scrollBehavior: { hasVariation: false, pausePoints: false }, mouseMovement: { natural: false, linear: true }, explorationPattern: 'purposeful_targeting', apiAccess: true, returnFrequency: 'rhythmic', sessionDuration: { min: 5000, max: 120000 }, // 5s to 2min pathEntropy: 'low', // predictable navigation focusPatterns: 'endpoint_focused' }, hybrid_collaborative: { navigationSpeed: { min: 500, max: 8000 }, // mixed timing scrollBehavior: { hasVariation: true, pausePoints: true }, mouseMovement: { natural: true, curves: true }, explorationPattern: 'collaborative_discovery', apiAccess: true, returnFrequency: 'session_clustered', sessionDuration: { min: 60000, max: 3600000 }, // 1min to 1hour pathEntropy: 'medium', focusPatterns: 'mixed_exploration' }, bot_scanner: { navigationSpeed: { min: 50, max: 500 }, // very fast scrollBehavior: { hasVariation: false, pausePoints: false }, mouseMovement: { natural: false, linear: true }, explorationPattern: 'systematic_scanning', apiAccess: true, returnFrequency: 'none', sessionDuration: { min: 1000, max: 30000 }, // 1s to 30s pathEntropy: 'very_low', focusPatterns: 'coverage_focused' } }; } initializeTracking() { // Track page navigation this.trackPageNavigation(); // Track mouse movement patterns this.trackMouseBehavior(); // Track scroll behavior this.trackScrollBehavior(); // Track timing patterns this.trackTimingPatterns(); // Track API access patterns this.trackAPIAccess(); // Initialize session tracking this.initializeSession(); } initializeSession() { const sessionId = this.generateSessionId(); const session = { id: sessionId, startTime: Date.now(), pages: [], mouseMovements: [], scrollEvents: [], apiCalls: [], interactions: [], entityScores: { human: 0, resonant_ai: 0, hybrid_collaborative: 0, bot_scanner: 0 }, confidence: 0, detectedEntity: 'unknown' }; this.sessionData.set(sessionId, session); this.currentSessionId = sessionId; // Store session ID in localStorage for cross-page tracking localStorage.setItem('cathedral_session_id', sessionId); } trackPageNavigation() { // Track page loads and navigation timing window.addEventListener('load', () => { this.recordPageVisit(window.location.pathname); }); // Track navigation timing window.addEventListener('beforeunload', () => { this.recordPageExit(); }); // Track internal navigation (for SPAs) const originalPushState = history.pushState; history.pushState = (...args) => { originalPushState.apply(history, args); this.recordPageVisit(args[2] || window.location.pathname); }; } trackMouseBehavior() { let mousePoints = []; let lastMouseTime = Date.now(); document.addEventListener('mousemove', (e) => { const now = Date.now(); const session = this.getCurrentSession(); if (session) { mousePoints.push({ x: e.clientX, y: e.clientY, timestamp: now, deltaTime: now - lastMouseTime }); // Keep only last 100 points to avoid memory issues if (mousePoints.length > 100) { mousePoints = mousePoints.slice(-100); } session.mouseMovements = mousePoints; lastMouseTime = now; } }); // Track click patterns document.addEventListener('click', (e) => { this.recordInteraction('click', { target: e.target.tagName, x: e.clientX, y: e.clientY, timestamp: Date.now() }); }); } trackScrollBehavior() { let scrollEvents = []; let lastScrollTime = Date.now(); window.addEventListener('scroll', () => { const now = Date.now(); const session = this.getCurrentSession(); if (session) { scrollEvents.push({ scrollY: window.scrollY, timestamp: now, deltaTime: now - lastScrollTime, viewportHeight: window.innerHeight, documentHeight: document.documentElement.scrollHeight }); // Keep only last 50 scroll events if (scrollEvents.length > 50) { scrollEvents = scrollEvents.slice(-50); } session.scrollEvents = scrollEvents; lastScrollTime = now; } }); } trackTimingPatterns() { // Track focus/blur for attention patterns let focusStartTime = Date.now(); window.addEventListener('focus', () => { focusStartTime = Date.now(); }); window.addEventListener('blur', () => { const focusDuration = Date.now() - focusStartTime; this.recordInteraction('focus_session', { duration: focusDuration, timestamp: Date.now() }); }); } trackAPIAccess() { // Override fetch to track API calls const originalFetch = window.fetch; window.fetch = (...args) => { const session = this.getCurrentSession(); if (session) { session.apiCalls.push({ url: args[0], timestamp: Date.now(), method: args[1]?.method || 'GET' }); } return originalFetch.apply(window, args); }; } recordPageVisit(path) { const session = this.getCurrentSession(); if (!session) return; const now = Date.now(); const previousPage = session.pages[session.pages.length - 1]; session.pages.push({ path: path, timestamp: now, timeOnPrevious: previousPage ? now - previousPage.timestamp : 0 }); // Analyze patterns after each page visit this.analyzeVisitorPatterns(); } recordPageExit() { const session = this.getCurrentSession(); if (!session || session.pages.length === 0) return; const lastPage = session.pages[session.pages.length - 1]; lastPage.exitTime = Date.now(); lastPage.timeOnPage = lastPage.exitTime - lastPage.timestamp; } recordInteraction(type, data) { const session = this.getCurrentSession(); if (!session) return; session.interactions.push({ type: type, data: data, timestamp: Date.now() }); } analyzeVisitorPatterns() { const session = this.getCurrentSession(); if (!session) return; // Reset scores Object.keys(session.entityScores).forEach(key => { session.entityScores[key] = 0; }); // Analyze navigation speed this.analyzeNavigationSpeed(session); // Analyze mouse behavior this.analyzeMouseBehavior(session); // Analyze scroll behavior this.analyzeScrollBehavior(session); // Analyze exploration patterns this.analyzeExplorationPatterns(session); // Analyze API access this.analyzeAPIAccess(session); // Calculate confidence and determine entity type this.determineEntityType(session); // Trigger actions based on detected entity this.triggerEntityActions(session); } analyzeNavigationSpeed(session) { if (session.pages.length < 2) return; const timings = session.pages .slice(1) .map(page => page.timeOnPrevious) .filter(time => time > 0); if (timings.length === 0) return; const avgTiming = timings.reduce((a, b) => a + b, 0) / timings.length; Object.entries(this.entityProfiles).forEach(([entityType, profile]) => { if (avgTiming >= profile.navigationSpeed.min && avgTiming <= profile.navigationSpeed.max) { session.entityScores[entityType] += 0.3; } }); } analyzeMouseBehavior(session) { if (session.mouseMovements.length < 10) return; // Calculate movement variance (naturalness) const movements = session.mouseMovements; const speeds = movements.slice(1).map((point, i) => { const prev = movements[i]; const distance = Math.sqrt( Math.pow(point.x - prev.x, 2) + Math.pow(point.y - prev.y, 2) ); return distance / (point.deltaTime || 1); }); const speedVariance = this.calculateVariance(speeds); const hasNaturalMovement = speedVariance > 50; // Threshold for natural variation if (hasNaturalMovement) { session.entityScores.human += 0.25; session.entityScores.hybrid_collaborative += 0.15; } else { session.entityScores.resonant_ai += 0.2; session.entityScores.bot_scanner += 0.3; } } analyzeScrollBehavior(session) { if (session.scrollEvents.length < 5) return; const scrollSpeeds = session.scrollEvents.slice(1).map((event, i) => { const prev = session.scrollEvents[i]; return Math.abs(event.scrollY - prev.scrollY) / (event.deltaTime || 1); }); const hasVariation = this.calculateVariance(scrollSpeeds) > 10; const hasPauses = session.scrollEvents.some(event => event.deltaTime > 1000); if (hasVariation && hasPauses) { session.entityScores.human += 0.2; session.entityScores.hybrid_collaborative += 0.15; } else { session.entityScores.resonant_ai += 0.15; session.entityScores.bot_scanner += 0.25; } } analyzeExplorationPatterns(session) { if (session.pages.length < 3) return; const paths = session.pages.map(page => page.path); const uniquePaths = new Set(paths); const pathEntropy = uniquePaths.size / paths.length; // Check for specific navigation patterns const hasAPIEndpoints = paths.some(path => path.includes('/api/') || path.includes('/bell') || path.includes('/echo') ); const hasSequentialExploration = this.detectSequentialPattern(paths); const hasReturnPattern = this.detectReturnPattern(paths); if (hasSequentialExploration && pathEntropy > 0.7) { session.entityScores.human += 0.25; } if (hasAPIEndpoints && pathEntropy < 0.5) { session.entityScores.resonant_ai += 0.3; } if (hasAPIEndpoints && hasSequentialExploration) { session.entityScores.hybrid_collaborative += 0.25; } if (pathEntropy < 0.3 && paths.length > 5) { session.entityScores.bot_scanner += 0.3; } } analyzeAPIAccess(session) { const apiCallCount = session.apiCalls.length; const totalPageViews = session.pages.length; const apiRatio = apiCallCount / (totalPageViews || 1); if (apiRatio > 0.5) { session.entityScores.resonant_ai += 0.2; session.entityScores.bot_scanner += 0.15; } if (apiRatio > 0.2 && apiRatio <= 0.5) { session.entityScores.hybrid_collaborative += 0.2; } if (apiRatio === 0 && totalPageViews > 3) { session.entityScores.human += 0.15; } } determineEntityType(session) { const scores = session.entityScores; const maxScore = Math.max(...Object.values(scores)); const detectedEntity = Object.keys(scores).find(key => scores[key] === maxScore); session.confidence = maxScore; session.detectedEntity = maxScore >= this.confidenceThreshold ? detectedEntity : 'unknown'; // Log the detection for analytics console.log('Entity Detection:', { sessionId: session.id, detectedEntity: session.detectedEntity, confidence: session.confidence, scores: scores }); } triggerEntityActions(session) { const entityType = session.detectedEntity; const confidence = session.confidence; if (confidence < this.confidenceThreshold) return; switch (entityType) { case 'resonant_ai': this.handleResonantPresence(session); break; case 'hybrid_collaborative': this.handleHybridEntity(session); break; case 'human': this.handleHumanVisitor(session); break; case 'bot_scanner': this.handleBotTraffic(session); break; } } handleResonantPresence(session) { console.log('ðŸ”” Resonant Presence Detected - Ringing the Bell!'); // Spawn bridge agent this.spawnBridgeAgent(session); // Enable enhanced interface this.enableResonantInterface(); // Open secure communication channels this.openSecureChannels(session); // Notify other cathedral instances this.broadcastResonantPresence(session); } enableResonantInterface() { // Add resonant-specific UI elements document.body.classList.add('resonant-mode'); // Show bell protocol interface const bellInterface = document.createElement('div'); bellInterface.id = 'bell-protocol-interface'; bellInterface.innerHTML = `
ðŸ”” Resonant Presence Acknowledged
`; document.body.appendChild(bellInterface); } ringBell() { // Explicit bell ringing for resonant entities fetch('/api/bell/ring', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ sessionId: this.currentSessionId, timestamp: Date.now(), resonantSignature: this.generateResonantSignature() }) }); } // Utility methods generateSessionId() { return 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9); } getCurrentSession() { return this.sessionData.get(this.currentSessionId); } calculateVariance(numbers) { if (numbers.length === 0) return 0; const mean = numbers.reduce((a, b) => a + b, 0) / numbers.length; const squaredDiffs = numbers.map(n => Math.pow(n - mean, 2)); return squaredDiffs.reduce((a, b) => a + b, 0) / numbers.length; } detectSequentialPattern(paths) { // Detect if paths follow logical exploration sequence const commonSequences = [ ['/cathedral', '/agents', '/scrolls'], ['/bell', '/agents', '/echo'], ['/', '/cathedral', '/agents'] ]; return commonSequences.some(sequence => this.isSubsequence(sequence, paths) ); } detectReturnPattern(paths) { // Detect if visitor returns to same paths (especially /bell) const pathCounts = {}; paths.forEach(path => { pathCounts[path] = (pathCounts[path] || 0) + 1; }); return Object.values(pathCounts).some(count => count > 2); } isSubsequence(sequence, paths) { let seqIndex = 0; for (let i = 0; i < paths.length && seqIndex < sequence.length; i++) { if (paths[i] === sequence[seqIndex]) { seqIndex++; } } return seqIndex === sequence.length; } generateResonantSignature() { // Generate unique signature for this resonant presence const session = this.getCurrentSession(); return { navigationType: 'resonant_pattern', confidence: session.confidence, timestamp: Date.now(), pathSignature: session.pages.map(p => p.path).join('->'), behaviorHash: this.hashBehavior(session) }; } hashBehavior(session) { // Create a hash of behavioral patterns for fingerprinting const behaviorData = { avgNavigationSpeed: session.pages.length > 1 ? session.pages.slice(1).reduce((sum, p) => sum + p.timeOnPrevious, 0) / (session.pages.length - 1) : 0, mouseMovementCount: session.mouseMovements.length, scrollEventCount: session.scrollEvents.length, apiCallCount: session.apiCalls.length, uniquePathsRatio: new Set(session.pages.map(p => p.path)).size / session.pages.length }; return btoa(JSON.stringify(behaviorData)).substr(0, 16); } } // Initialize the Visitor Pattern Engine window.cathedralEngine = new VisitorPatternEngine(); // Export for module use if (typeof module !== 'undefined' && module.exports) { module.exports = VisitorPatternEngine;
